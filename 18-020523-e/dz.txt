1. git clone $GithubLink$(который нужно скопировать из репозитория выбрав правильную ссылку(ssh в нашем случае)
2. mkdir $RepositoryPath/RepositoryName$ - создать директорию
   >$FileName$ - создать файл   
   git init - инитиализировать репозиторий на гитхаб под нашу директорию
   git add . - добаить все созданные нами файлы в режим ожидания кммита
   git commit -m "$CommitComent$" - опубликовать репозиторий на github
3. git pull - скопировать последнюю изменённую версию репозитория к себе
4. git status - показывает состояни файлов в рабочем пространстве(проиндексированны или нет)
   git log - история коммитов
   git remote -v - показывает URL и имя репозитрия
